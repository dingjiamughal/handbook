- 手写 bind

```js
Function.prototype.bind = function (ctx = window, ...args) {
  let _this = this;

  return function F(...innerArgs) {
    if (_this instanceof F) {
      return new _this(...args, ...innerArgs);
    }

    return _this.apply(ctx, [...args, ...innerArgs]);
  };
};
```

- interface 和 type 的区别

```js
// interface
interface X { x: number; }
interface Y extends X { y: number; }
type Y = X & { y: number; };

// interface - implements
interface X {x: number; y: number;}

class ClassX implements X {
  x = 1;
  y = 2;
}

// interface - merge
interface X {x: number;}
interface X {y: number;}
const a: X = {x:1, y:2 };

// type
type 可以用于基础类型、联合和元祖
type X = { x: number; };
type Y = X & { y: number; };
interface Y extends X { y: number; }
```
